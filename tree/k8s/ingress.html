# Ingress

# 创建Ingress

创建nginx-ingress.yaml文件

apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: nginx-ingress
spec:
  ingressClassName: nginx
  defaultBackend:
    service:
      name: saber-service
      port:
        number: 80
  rules:
  - host: bladex-dev.ptii.com
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: saber-service
            port:
              number: 80
