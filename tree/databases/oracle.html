# Oracle

# 常用操作

-- 创建表空间
create tablespace MAMWC 
-- 表空间物理文件名称
datafile 'MAMWC.dbf'
-- 这种方式指定表空间物理文件位置
-- datafile 'F:\APP\QIUCHANGJIN\ORADATA\ORCL\QCJ_TABLESPACE.dbf' 
-- 大小 100M，每次 5M 自动增大，最大不限制
size 100M autoextend on next 5M maxsize unlimited; 


-- 创建用户
create user mamwc IDENTIFIED BY mamwc --用户密码
default tablespace MAMWC -- 表空间是上面创建的
temporary tablespace TEMP -- 临时表空间默认 TEMP
profile DEFAULT;
-- password expire;
-- 密码过期需要重设密码,意思是当你用这个新建立的密码过期用户首次登录后，系统会提示你重新输入新密码，不然会拒绝你登陆，重设新密码后就可以登录，该用户的密码就是你新设的密码，相当于首次登录修改密码这样的选项。

-- 用户授权_1
grant connect,resource,dba to mamwc;
-- 用户授权_2
grant connect to mamwc with admin option;
grant dba to mamwc with admin option;
grant resource to mamwc with admin option;
